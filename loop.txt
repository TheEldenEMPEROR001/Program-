def main():
    while True:  # Loop the game menu
        print("\n=====================================")
        print("      Welcome to the Detective Game")
        print("=====================================")
        
        show_cases()  # show available cases
        
        try:
            choice = int(input("\nEnter the case number to play (0 to exit): "))
            
            if choice == 0:
                print("Thanks for playing! Goodbye ")
                break  # exits the loop and ends the program
            
            play_case(choice)  # play the chosen case
            
            # After playing a case, ask what to do next
            while True:
                print("\nWhat would you like to do next?")
                print("1. Play another case")
                print("2. Replay this case")
                print("3. Exit game")
                next_action = input("Enter your choice: ")

                if next_action == "1":
                    break  # go back to the main menu
                elif next_action == "2":
                    play_case(choice)  # replay same case
                elif next_action == "3":
                    print("Thanks for playing! Goodbye ")
                    return  # completely stop the game
                else:
                    print("Invalid option. Try again.")
                    
        except ValueError:
            print("Please enter a valid number.")
